

function checkAccountuser($username, $password) {
        // Open database connection
        $con = $this->opencon();
    
        // Prepare the SQL query
        $stmt = $con->prepare("SELECT * FROM customer WHERE username = ?");
        $stmt->execute([$username]);
    
        // Fetch the user data as an associative array
        $user = $stmt->fetch(PDO::FETCH_ASSOC);
    
        // If a user is found, verify the password
        if ($user && password_verify($password, $user['admin_pass'])) {
            return $user;
        }
    
        // If no user is found or password is incorrect, return false
        return false;
    }

function checkAccountAdmin($username, $password) {
    // Open database connection
    $con = $this->opencon();
    
    // Prepare the SQL query
    $stmt = $con->prepare("SELECT * FROM admin WHERE username = ?");
    $stmt->execute([$username]);
    
    // Fetch the admin data as an associative array
    $admin = $stmt->fetch(PDO::FETCH_ASSOC);
    
    // If an admin is found, verify the password
    if ($admin && password_verify($password, $admin['password'])) {
        return $admin;
    }
    
    // If no admin is found or password is incorrect, return false
    return false;
}
